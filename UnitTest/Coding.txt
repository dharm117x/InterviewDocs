1. Static mock:
----------------
public class DateUtils {
    public static LocalDate today() {
        return LocalDate.now();
    }
}

public class ReportService {
    public String generateReportDateString() {
        return "Report Date: " + DateUtils.today().toString();
    }
}

try (MockedStatic<DateUtils> mockedStatic = mockStatic(DateUtils.class)) {

            // Define the static method's mock behavior
            mockedStatic.when(DateUtils::today).thenReturn(LocalDate.of(2023, 12, 25));

            // Create service and call method
            ReportService service = new ReportService();
            String result = service.generateReportDateString();

            // Assert
            assertEquals("Report Date: 2023-12-25", result);

            // Verify static method was called
            mockedStatic.verify(DateUtils::today);
}
