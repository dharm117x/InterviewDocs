Step 1. Create new extension with template ycomercewebservice or use it exsiting one.

Step 2. Create controller extends with BaseCommerceController.

Step 3. Create WSDTO beans in extension_name-beans.xml for response obj.

Step 4. Mapping Data to WSDTO in dto-maaings-v2-spring.xml
	<bean id="studentSearchPageFieldMapper" parent="fieldMapper">
		<property name="sourceClass"
		          value="com.dkshopocc.data.student.StudentSearchPageData"/>
		<property name="destClass"
		          value="com.dkshopocc.dto.student.StudentSearchPageWsDTO"/>
		<property name="fieldMapping">
			<map>
				<entry key="results" value="students"/>
			</map>
		</property>
	</bean>
	


Step 5. Define scope(BASIC,DEFAULT,FULL) for WSDTO object in dto-level-mappings-v2-spring.xml
	<bean parent="fieldSetLevelMapping"	id="studentDataListWSDTOFieldSetLevelMapping">
		<property name="dtoClass" value="com.dkshopocc.dto.student.StudentDataListWSDTO" />
		<property name="levelMapping">
            <map>
                <entry key="BASIC" value="students"/>
                <entry key="DEFAULT" value="students(DEFAULT)"/>
                <entry key="FULL" value="students(FULL)"/>
            </map>
        </property>
	</bean>

	<bean parent="fieldSetLevelMapping"	id="studentDataWSDTOFieldSetLevelMapping">
		<property name="dtoClass" value="com.dkshopocc.dto.student.StudentDataWSDTO" />
		<property name="levelMapping">
			<map>
				<entry key="BASIC" value="studentId,studentName" />
				<entry key="DEFAULT" value="BASIC,studentPlace" />
				<entry key="FULL" value="DEFAULT,studentGuardian" />
			 </map>
        </property>
	</bean>
